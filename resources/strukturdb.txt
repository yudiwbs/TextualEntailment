CREATE TABLE IF NOT EXISTS `disc_t` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `id_kalimat` int(11) DEFAULT NULL,
  `t` text,
  `t_gram_structure` text,
  `t_type_dependency` text,
  `t_subject` text,
  `t_verb` text,
  `t_obj` text,
